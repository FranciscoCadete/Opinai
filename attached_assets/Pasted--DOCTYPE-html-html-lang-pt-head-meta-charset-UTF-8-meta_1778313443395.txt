<script>
/* ── ANIMATED BACKGROUND PARTICLES ── */
const canvas = document.getElementById('bg-canvas');
const ctx    = canvas.getContext('2d');

let W, H, particles, lines;

function resize() {
  W = canvas.width  = window.innerWidth;
  H = canvas.height = window.innerHeight;
}

class Particle {
  constructor() { this.reset(); }
  reset() {
    this.x  = Math.random() * W;
    this.y  = Math.random() * H;
    this.vx = (Math.random() - 0.5) * 0.18;
    this.vy = (Math.random() - 0.5) * 0.18;
    this.r  = Math.random() * 1.2 + 0.3;
    this.a  = Math.random() * 0.5 + 0.1;
  }
  update() {
    this.x += this.vx;
    this.y += this.vy;
    if (this.x < 0 || this.x > W) this.vx *= -1;
    if (this.y < 0 || this.y > H) this.vy *= -1;
  }
  draw() {
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(0,196,154,${this.a})`;
    ctx.fill();
  }
}

function init() {
  resize();
  particles = Array.from({ length: 70 }, () => new Particle());
}

function drawLines() {
  for (let i = 0; i < particles.length; i++) {
    for (let j = i + 1; j < particles.length; j++) {
      const dx   = particles[i].x - particles[j].x;
      const dy   = particles[i].y - particles[j].y;
      const dist = Math.sqrt(dx*dx + dy*dy);
      if (dist < 120) {
        const a = (1 - dist / 120) * 0.08;
        ctx.beginPath();
        ctx.moveTo(particles[i].x, particles[i].y);
        ctx.lineTo(particles[j].x, particles[j].y);
        ctx.strokeStyle = `rgba(0,196,154,${a})`;
        ctx.lineWidth = 0.5;
        ctx.stroke();
      }
    }
  }
}

function animate() {
  ctx.clearRect(0, 0, W, H);
  particles.forEach(p => { p.update(); p.draw(); });
  drawLines();
  requestAnimationFrame(animate);
}

window.addEventListener('resize', () => { resize(); });
init();
animate();

/* ── ROLE SELECTOR ── */
let selectedRole = 'admin';

function selectRole(el) {
  document.querySelectorAll('.role-opt').forEach(o => o.classList.remove('selected'));
  el.classList.add('selected');
  selectedRole = el.dataset.role;
}

/* ── TOGGLE PASSWORD ── */
let passVisible = false;
function togglePass() {
  passVisible = !passVisible;
  const inp = document.getElementById('pass-input');
  inp.type  = passVisible ? 'text' : 'password';
  document.getElementById('eye-icon').innerHTML = passVisible
    ? '<path d="M17.94 17.94A10.07 10.07 0 0 1 12 20c-7 0-11-8-11-8a18.45 18.45 0 0 1 5.06-5.94M9.9 4.24A9.12 9.12 0 0 1 12 4c7 0 11 8 11 8a18.5 18.5 0 0 1-2.16 3.19M6.53 6.53 17.47 17.47"/><path d="M10.73 10.73a3 3 0 0 0 4.24 4.24"/>'
    : '<path d="M1 12s4-8 11-8 11 8 11 8-4 8-11 8-11-8-11-8Z"/><circle cx="12" cy="12" r="3"/>';
}

/* ── VALIDATION ── */
function validate() {
  let ok = true;
  const email = document.getElementById('email-input');
  const pass  = document.getElementById('pass-input');
  const emailErr = document.getElementById('email-error');
  const passErr  = document.getElementById('pass-error');

  email.classList.remove('error');
  pass.classList.remove('error');
  emailErr.classList.remove('show');
  passErr.classList.remove('show');

  if (!email.value || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.value)) {
    email.classList.add('error');
    emailErr.classList.add('show');
    ok = false;
  }
  if (!pass.value || pass.value.length < 4) {
    pass.classList.add('error');
    passErr.classList.add('show');
    ok = false;
  }
  return ok;
}

/* ── HANDLE LOGIN ── */
function handleLogin() {
  if (!validate()) return;

  const btn = document.getElementById('submit-btn');
  const errBanner = document.getElementById('error-banner');
  errBanner.classList.remove('show');

  btn.classList.add('loading');

  // Simulate API call
  setTimeout(() => {
    btn.classList.remove('loading');

    // Demo: admin@mulenvos.ao / op1na1 = success, else error
    const email = document.getElementById('email-input').value;
    const pass  = document.getElementById('pass-input').value;

    if (email === 'admin@mulenvos.ao' && pass === 'op1na1') {
      showSuccess();
    } else {
      document.getElementById('error-msg').textContent =
        'Credenciais inválidas. Tente: admin@mulenvos.ao / op1na1';
      errBanner.classList.add('show');
    }
  }, 1800);
}

/* ── SUCCESS ── */
function showSuccess() {
  document.getElementById('form-inner').classList.add('hidden');
  const overlay = document.getElementById('success-overlay');
  overlay.classList.add('show');

  setTimeout(() => {
    const fill = document.getElementById('redirect-fill');
    fill.style.width = '100%';
  }, 100);

  setTimeout(() => {
    // In production: window.location.href = '/dashboard';
    alert('Redirecionando para o Dashboard do ' + selectedRole.charAt(0).toUpperCase() + selectedRole.slice(1) + '…\n\n(Em produção: window.location.href = "/dashboard")');
    document.getElementById('form-inner').classList.remove('hidden');
    overlay.classList.remove('show');
    document.getElementById('redirect-fill').style.transition = 'none';
    document.getElementById('redirect-fill').style.width = '0';
    setTimeout(() => {
      document.getElementById('redirect-fill').style.transition = 'width 2.5s linear';
    }, 50);
  }, 3000);
}
