/* ── ANIMATE STATS ON LOAD ── */
function animateCounter(el, target, duration, suffix) {
  let start = 0;
  const step = target / (duration / 16);
  const timer = setInterval(() => {
    start = Math.min(start + step, target);
    el.textContent = Math.round(start) + (suffix || '');
    if (start >= target) clearInterval(timer);
  }, 16);
}

window.addEventListener('load', () => {
  setTimeout(() => {
    animateCounter(document.getElementById('stat-resolved'), 714, 1200);
    animateCounter(document.getElementById('stat-open'), 389, 1000);
    animateCounter(document.getElementById('stat-time'), 38, 900, 'h');
    animateCounter(document.getElementById('stat-mediators'), 3, 600);
  }, 300);
});

/* ── GPS ── */
function getGPS() {
  const btn = document.getElementById('gps-btn');
  const lbl = document.getElementById('gps-label');
  btn.style.opacity = '0.7';
  lbl.textContent = 'A obter localização...';

  if (!navigator.geolocation) {
    lbl.textContent = 'GPS não disponível neste dispositivo.';
    btn.style.opacity = '1';
    return;
  }

  navigator.geolocation.getCurrentPosition(
    (pos) => {
      state.gpsCoords = { lat: pos.coords.latitude, lng: pos.coords.longitude };
      btn.classList.add('got');
      btn.style.opacity = '1';
      lbl.textContent = `GPS obtido: ${pos.coords.latitude.toFixed(4)}, ${pos.coords.longitude.toFixed(4)}`;
      showToast('Localização GPS capturada com sucesso!');
    },
    () => {
      lbl.textContent = 'Não foi possível obter GPS. Seleccione o bairro manualmente.';
      btn.style.opacity = '1';
    }
  );
}

/* ── FILE UPLOAD ── */
function handleFiles(input) {
  const files = Array.from(input.files);
  const previews = document.getElementById('upload-previews');
  previews.innerHTML = '';
  state.files = files.slice(0, 3);

  state.files.forEach(file => {
    if (!file.type.startsWith('image/')) return;
    const reader = new FileReader();
    reader.onload = (e) => {
      const wrap = document.createElement('div');
      wrap.className = 'preview-thumb-wrap';
      const img = document.createElement('img');
      img.src = e.target.result;
      img.className = 'preview-thumb';
      const rm = document.createElement('button');
      rm.className = 'preview-remove';
      rm.textContent = '×';
      rm.onclick = () => { wrap.remove(); };
      wrap.appendChild(img);
      wrap.appendChild(rm);
      previews.appendChild(wrap);
    };
    reader.readAsDataURL(file);
  });

  document.getElementById('upload-area').addEventListener('dragover', (e) => { e.preventDefault(); document.getElementById('upload-area').classList.add('drag-over'); });
  document.getElementById('upload-area').addEventListener('dragleave', () => { document.getElementById('upload-area').classList.remove('drag-over'); });
}

  const counter = document.getElementById(countId);

  counter.className = 'char-count' + (len > max * 0.9 ? ' near' : '') + (len >= max ? ' over' : '');
}

/* ── FIELD VALIDATION ── */
function validateField(el) {
  const id = el.id;
  const errEl = document.getElementById(id + '-err');
  if (errEl) {
    const valid = el.value && el.value.trim().length > 0;
    el.classList.toggle('err', !valid);
    errEl.classList.toggle('show', !valid);
    return valid;
  }
  return true;
}

  if (thumbs.length > 0) {
    imgRow.style.display = 'flex';
    thumbs.forEach(img => {
      const cl = document.createElement('img');
      cl.src = img.src;
      cl.className = 'confirm-thumb';
      imgCont.appendChild(cl);
    });
  } else {
    imgRow.style.display = 'none';
  }
}

/* ── SUBMIT ── */
function submitForm() {
  if (!state.termsAccepted) {
    document.getElementById('terms-err').classList.add('show');
    return;
  }

  const btn = document.getElementById('submit-btn');
  btn.classList.add('loading');

  // Simulate API call
  setTimeout(() => {
    btn.classList.remove('loading');
    const now = new Date();
    const pad = n => String(n).padStart(2,'0');
    const dateStr = `${now.getFullYear()}${pad(now.getMonth()+1)}${pad(now.getDate())}`;
    const rand = String(Math.floor(Math.random() * 9000) + 1000);
    const ticketId = `MUL-${dateStr}-${rand}`;
    state.ticketId = ticketId;
    document.getElementById('success-ticket-id').textContent = ticketId;
    showSuccess();
  }, 2200);
}

function trackTicket() {
  const val = document.getElementById('track-input').value.trim();
  if (!val) return;
  const card = document.getElementById('tsc');
  card.classList.add('show');

  document.getElementById('tsc-id').textContent = val.toUpperCase();

  const demos = [
    { desc: 'Falta de água na rua principal', steps: [
      { label: 'Pedido recebido', done: true, time: 'Ontem 09:14' },
      { label: 'Em triagem (IA)', done: true, time: 'Ontem 09:15' },
      { label: 'Atribuído ao técnico', done: true, time: 'Ontem 11:30' },
      { label: 'Em progresso', done: false, current: true, time: 'Hoje 08:00' },
      { label: 'Resolvido', done: false, time: '—' }
    ]},
    { desc: 'Recolha de lixo — em falta há 3 dias', steps: [
      { label: 'Pedido recebido', done: true, time: '2 dias atrás 14:22' },
      { label: 'Em triagem (IA)', done: true, time: '2 dias atrás 14:22' },
      { label: 'Atribuído ao técnico', done: true, time: '2 dias atrás 16:10' },
      { label: 'Em progresso', done: true, time: 'Ontem 09:00' },
      { label: 'Resolvido', done: false, current: true, time: 'Hoje' }
    ]}
  ];

  const demo = demos[Math.floor(Math.random() * demos.length)];
  document.getElementById('tsc-desc').textContent = demo.desc;

  const tl = document.getElementById('tsc-timeline');
  tl.innerHTML = '';

  demo.steps.forEach((step, i) => {
    const item = document.createElement('div');
    item.className = 'tl-item';

    const wrap = document.createElement('div');
    wrap.className = 'tl-line-wrap';

    const dot = document.createElement('div');
    dot.className = 'tl-dot' + (step.done ? ' done' : step.current ? ' current' : '');
    wrap.appendChild(dot);

    if (i < demo.steps.length - 1) {
      const conn = document.createElement('div');
      conn.className = 'tl-connector' + (step.done ? ' done' : '');
      wrap.appendChild(conn);
    }

    const content = document.createElement('div');
    content.className = 'tl-content';
    content.innerHTML = `<div class="tl-status ${step.done ? 'done' : step.current ? 'current' : ''}">${step.label}</div><div class="tl-time">${step.time}</div>`;

    item.appendChild(wrap);
    item.appendChild(content);
    tl.appendChild(item);
  });
}

  }
}

}
