<script>
  // Animate bars on load
  window.addEventListener('load', () => {
    const maxVal = 92;
    document.querySelectorAll('.bar-fill').forEach(bar => {
      const val = parseInt(bar.dataset.val);
      const pct = (val / maxVal) * 100;
      setTimeout(() => { bar.style.width = pct + '%'; }, 100);
    });

    // SLA bar
    const slaBar = document.getElementById('sla-bar');
    if (slaBar) {
      setTimeout(() => { slaBar.style.width = slaBar.dataset.val + '%'; }, 200);
    }
  });

  // Live countdown
  let countdown = 30;
  const countEl = document.getElementById('countdown');
  const lastEl  = document.getElementById('last-update');

  setInterval(() => {
    countdown--;
    if (countdown <= 0) {
      countdown = 30;
      lastEl.textContent = new Date().toLocaleString('pt-AO');
    }
    countEl.textContent = countdown + 's';
  }, 1000);

  // Feed timer display
  const feedTimer = document.getElementById('feed-timer');
  let feedCd = 30;
  setInterval(() => {
    feedCd--;
    if (feedCd <= 0) feedCd = 30;
    feedTimer.textContent = 'Actualizar em ' + feedCd + 's';
  }, 1000);

  // Nav items toggle
  document.querySelectorAll('.nav-item').forEach(item => {
    item.addEventListener('click', e => {
      e.preventDefault();
      document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
      item.classList.add('active');
    });
  });

  // Card buttons toggle
  document.querySelectorAll('.card-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      const siblings = btn.closest('.card-actions').querySelectorAll('.card-btn');
      siblings.forEach(s => s.classList.remove('active'));
      btn.classList.add('active');
    });
  });
</script>
